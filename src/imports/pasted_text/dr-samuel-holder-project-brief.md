1. 🖼️ VISUAL BONITO E PROFISSIONAL:
   ✨ Design deve impressionar visualmente
   ✨ Usar as cores fornecidas de forma harmoniosa (70-25-5 rule)
      - 70% cor dominante (geralmente neutra/background)
      - 25% cor secundária
      - 5% cor de destaque (CTAs, destaques)
   ✨ Gradientes sutis e sombras suaves (Tailwind)
   ✨ Bordas arredondadas modernas (rounded-lg, rounded-xl, rounded-2xl)
   ✨ Glassmorphism ou efeitos modernos onde apropriado
   ✨ Hover states suaves em todos os elementos interativos
   ✨ Microinterações (botões que "reagem" ao hover)
   ✨ Consistência visual em TODOS os componentes

2. 📐 ESPAÇAMENTO IMPECÁVEL:
   ✅ Usar escala de espaçamento do Tailwind (4, 8, 12, 16, 24, 32, 48, 64px)
   ✅ Respiração generosa entre seções:
      - Mobile: py-12 ou py-16
      - Desktop: py-16 md:py-24 lg:py-32
   ✅ Padding consistente em containers:
      - Mobile: px-4 sm:px-6
      - Desktop: md:px-8 lg:px-16 xl:px-24
   ✅ Espaçamento entre elementos relacionados:
      - Tight: gap-2 (elementos muito próximos)
      - Normal: gap-4 ou gap-6 (padrão)
      - Generous: gap-8 ou gap-12 (seções)
   ✅ Max-width containers para conteúdo:
      - Padrão: max-w-7xl mx-auto
      - Texto: max-w-3xl mx-auto (leitura confortável)
      - Full: max-w-full (quando necessário)
   ✅ NUNCA elementos "colados" - sempre respiro visual
   ✅ Regra do 8px: todos os espaçamentos múltiplos de 8

3. 🔤 TIPOGRAFIA PROFISSIONAL:
   ✅ Hierarquia clara e consistente:
      - H1: Título principal (apenas 1 por página)
      - H2: Títulos de seção
      - H3: Subtítulos
      - Body: Texto corrido
      - Small: Textos secundários
   
   ✅ Tamanhos responsivos (scale system):
      H1:
      - Mobile: text-4xl (36px)
      - Tablet: md:text-5xl (48px)
      - Desktop: lg:text-6xl xl:text-7xl (60-72px)
      
      H2:
      - Mobile: text-3xl (30px)
      - Tablet: md:text-4xl (36px)
      - Desktop: lg:text-5xl (48px)
      
      H3:
      - Mobile: text-2xl (24px)
      - Desktop: md:text-3xl (30px)
      
      Body:
      - Mobile: text-base (16px)
      - Desktop: lg:text-lg (18px)
      
      Small:
      - text-sm (14px) ou text-xs (12px)
   
   ✅ Font weights estratégicos:
      - Títulos: font-bold (700) ou font-extrabold (800)
      - Subtítulos: font-semibold (600)
      - Body: font-normal (400)
      - Destaques: font-medium (500)
      - Secundário: font-light (300)
   
   ✅ Line-height (leading):
      - Títulos: leading-tight (1.25) ou leading-snug (1.375)
      - Body: leading-relaxed (1.625) ou leading-loose (2)
   
   ✅ Letter-spacing (tracking):
      - Títulos grandes: tracking-tight
      - Títulos pequenos: tracking-normal
      - UPPERCASE: tracking-wide ou tracking-wider
   
   ✅ Cores de texto com contraste:
      - Primário: text-gray-900 ou text-slate-900
      - Secundário: text-gray-600 ou text-slate-600
      - Terciário: text-gray-500

```typescript
// src/components/common/Button.tsx

/**
 * Componente de botão reutilizável
 * 
 * Usado em CTAs, formulários e ações importantes do site.
 * Suporta três variantes visuais e três tamanhos diferentes.
 * Todos os estados (hover, focus, disabled) estão implementados.
 * 
 * @example
 * <Button variant="primary" size="lg" onClick={handleCTA}>
 *   Começar Agora
 * </Button>
 */

import { ReactNode } from 'react';

// ════════════════════
// TYPES E INTERFACES
// ════════════════════

interface ButtonProps {
  children: ReactNode;                                    // Conteúdo do botão
  variant?: 'primary' | 'secondary' | 'outline';         // Estilo visual
  size?: 'sm' | 'md' | 'lg';                             // Tamanho do botão
  onClick?: () => void;                                   // Função ao clicar
  type?: 'button' | 'submit' | 'reset';                  // Tipo HTML
  disabled?: boolean;                                     // Estado desabilitado
  fullWidth?: boolean;                                    // Largura total
  icon?: ReactNode;                                       // Ícone opcional
}

// ══════════════════════
// COMPONENTE
// ══════════════════════

export const Button = ({ 
  children, 
  variant = 'primary',
  size = 'md',
  onClick,
  type = 'button',
  disabled = false,
  fullWidth = false,
  icon
}: ButtonProps) => {
  
  // ───────────────────────
  // CLASSES CSS (Tailwind)
  // ───────────────────────
  
  // Classes base - sempre presentes em qualquer botão
  const baseClasses = `
    inline-flex items-center justify-center gap-2
    rounded-lg font-semibold
    transition-all duration-300
    focus:outline-none focus:ring-2 focus:ring-offset-2
    disabled:opacity-50 disabled:cursor-not-allowed
  `;
  
  // Classes de variante - define a aparência visual
  const variantClasses = {
    primary: `
      bg-primary-600 hover:bg-primary-700 active:bg-primary-800
      text-white
      shadow-md hover:shadow-lg
      focus:ring-primary-500
    `,
    secondary: `
      bg-secondary-600 hover:bg-secondary-700 active:bg-secondary-800
      text-white
      shadow-md hover:shadow-lg
      focus:ring-secondary-500
    `,
    outline: `
      border-2 border-primary-600
      text-primary-600 hover:text-white
      hover:bg-primary-600
      focus:ring-primary-500
    `
  };
  
  // Classes de tamanho - padding e texto
  const sizeClasses = {
    sm: 'px-4 py-2 text-sm',
    md: 'px-6 py-3 text-base',
    lg: 'px-8 py-4 text-lg'
  };
  
  // Largura total (opcional)
  const widthClass = fullWidth ? 'w-full' : '';
  
  // Efeito de escala ao clicar (apenas se não estiver desabilitado)
  const activeClass = !disabled ? 'active:scale-95' : '';

  // ───────────────────────
  // RENDER
  // ──────────────────────
  
  return (
    <button
      type={type}
      onClick={onClick}
      disabled={disabled}
      className={`
        ${baseClasses}
        ${variantClasses[variant]}
        ${sizeClasses[size]}
        ${widthClass}
        ${activeClass}
      `.trim().replace(/\s+/g, ' ')}  // Remove espaços extras
    >
      {icon && <span className="flex-shrink-0">{icon}</span>}
      {children}
    </button>
  );
};
```
